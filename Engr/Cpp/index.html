Problem No. 3

Basic Strings, you should do as many examples organizing tables and such if you want to get good at strings.


Here's some code: 

 
                #include 
                    using std::cout;
                    using std::cin;
                    
                    #include 
                    using std::string;
                    
                    
                    int main()
                    {
                        string name;
                        cout << "Who are you?" << '\n';
                        cin >> name;
                    
                        string Greeting = "Hello, " + name;
                    
                        if (name == "Kate")
                        {
                            Greeting += ", I know you!";
                        } 
                            cout << Greeting << '\n';
                        return 0;
                    }
                
                Who are you?
                Kate
                Hello, Kate, I know you!
